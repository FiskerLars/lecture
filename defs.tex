\newcommand{\protocolbrief}[4]{
\begin{center}
        \cornersize*{10pt}
        \Ovalbox{
          \parbox{9.95cm}{\hspace{-3mm}
            \begin{tabular}{l|l}
              \multicolumn{2}{p{10cm}}{\textsf{\large \bf {#1}}}\\\hline
              \textsf{\large{#2}}& \\\hline
              \multicolumn{2}{p{10cm}}{\textsf{\large{#3}}}\\\hline
              \multicolumn{2}{p{10cm}}{\textsf{\large{#4}}}
            \end{tabular}
          }
        }
\end{center}
\vspace{\baselineskip}
}
